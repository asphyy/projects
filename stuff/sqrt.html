let x = x;

let nps = nearestPerfectSquare(x).squareRoot;                  //probably by looking through a pre-made table of integers squared

for (i = 1; i < iterations; i++){

    a = ((2*x / ((a / 2) + (x*(x - Math.floor(x))) / a))+a)/2   //averages 2*(guess of square root of x with new guess of a) and the last value of a

}

let sqrt = x / ((a / 2) + (x*(x - Math.floor(x))) / a)          //final iteration
